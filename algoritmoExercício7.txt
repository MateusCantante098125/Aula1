Entrada: número
Saída : algarismo_maior

Início

# Ler o número
Escrever "Introduza um número : "

Função alg_maior (número)
    algarismo_maior <- "0"
    # Calcular algorismo maior
    para n in número faz
        se n > algarismo_maior então :
            algarismo_maior <- n
        fim se

# Imprimir Resultado
Escrever "O algarismo mais alto do número",número,"é",algarismo_maior
